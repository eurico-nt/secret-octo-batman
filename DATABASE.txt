
-----CRIAR BANCO DE DADOS
create database projectlp
---criar tabelas
CREATE TABLE empresas(
empid int PRIMARY KEY NOT NULL UNIQUE,
nome varchar(20),
responsavel varchar(20)
)
----tabela decisoes

CREATE TABLE decisoes
(
  codigo integer NOT NULL,
  periodo integer,
  preco double precision,
  marketing double precision,
  producao integer,
  quantidade integer,
  codigo_empresa integer,
  CONSTRAINT decisoes_pkey PRIMARY KEY (codigo),
  CONSTRAINT decisoes_codigo_empresa_fkey FOREIGN KEY (codigo_empresa)
      REFERENCES empresas (empid) MATCH SIMPLE
      ON UPDATE NO ACTION ON DELETE NO ACTION
)
